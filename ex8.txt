🔹 EXPERIMENT NO. 8
Intermediate Code Generation (Three Address Code)
________________________________________
🎯 1. Objective
•	To generate Intermediate Code 
•	To convert high-level expressions into Three Address Code (TAC) 
•	To understand how compiler simplifies expressions 
________________________________________
📘 2. Theory (Simple Explanation)
🔹 What is Intermediate Code?
•	It is machine-independent code 
•	Generated between: 
o	Source Code → Machine Code 
________________________________________
🔹 What is Three Address Code (TAC)?
•	Each instruction has at most 3 operands 
👉 Format:
result = operand1 op operand2
________________________________________
✅ Example
Input Expression:
a = b + c * d
Output (TAC):
t1 = c * d
t2 = b + t1
a = t2

/* icg.y */
%{
#include<stdio.h>
#include<stdlib.h>
#include<string.h>

int temp = 1;

void yyerror(char *s);
int yylex();

char* newTemp()
{
    static char t[10];
    sprintf(t, "t%d", temp++);
    return strdup(t);
}
%}

%union{
    char str[20];
}

%token <str> ID NUM
%type  <str> expr

%%

input:
      stmt '\n'
      ;

stmt:
      ID '=' expr
      {
          printf("%s = %s\n", $1, $3);
      }
      ;

expr:
      expr '+' expr
      {
          char *t = newTemp();
          printf("%s = %s + %s\n", t, $1, $3);
          strcpy($$, t);
      }
    | expr '*' expr
      {
          char *t = newTemp();
          printf("%s = %s * %s\n", t, $1, $3);
          strcpy($$, t);
      }
    | ID   { strcpy($$, $1); }
    | NUM  { strcpy($$, $1); }
    ;

%%

void yyerror(char *s)
{
    printf("Invalid Input\n");
}

int main()
{
    printf("Enter Expression:\n");
    yyparse();
    return 0;
}________________________________________
