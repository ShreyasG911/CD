Implement Symbol Table using Different Data Structures

🎯 1. Objective
To understand:

________________________________________
📘 2. Theory (Simple Explanation)
A Symbol Table is a data structure used by a compiler to store information about:
•	Variables
•	Functions
•	Constants
•	Identifiers
It stores details like:
Name	Type	Scope	Value
________________________________________
🔧 Operations on Symbol Table
1.	Insert() → Add new identifier
2.	Lookup() → Search identifier
3.	Display() → Show symbol table
________________________________________
🗂 Data Structures Used
1.	Linear List (Array) ✅ (Easy – mostly used in practical)
2.	Hash Table
3.	Binary Search Tree
👉 Here we implement using Linear List (Array).

💻 3. C Program – Symbol Table using Array
#include <stdio.h>
#include <string.h>

#define MAX 100

// Structure of Symbol Table
struct symbol
{
    char name[50];
    char type[20];
    char scope[20];
};

struct symbol table[MAX];
int count = 0;

// Insert function
void insert()
{
    printf("Enter Identifier Name: ");
    scanf("%s", table[count].name);

    printf("Enter Data Type: ");
    scanf("%s", table[count].type);

    printf("Enter Scope (Global/Local): ");
    scanf("%s", table[count].scope);

    count++;
    printf("Identifier Inserted Successfully!\n");
}

// Lookup function
void lookup()
{
    char search[50];
    int i, found = 0;

    printf("Enter Identifier to Search: ");
    scanf("%s", search);

    for(i = 0; i < count; i++)
    {
        if(strcmp(search, table[i].name) == 0)
        {
            printf("Identifier Found!\n");
            printf("Name: %s\nType: %s\nScope: %s\n",
                   table[i].name, table[i].type, table[i].scope);
            found = 1;
            break;
        }
    }

    if(found == 0)
        printf("Identifier Not Found!\n");
}

// Display function
void display()
{
    int i;

    printf("\n----- SYMBOL TABLE -----\n");
    printf("Name\tType\tScope\n");

    for(i = 0; i < count; i++)
    {
        printf("%s\t%s\t%s\n",
               table[i].name,
               table[i].type,
               table[i].scope);
    }
}

int main()
{
    int choice;

    while(1)
    {
        printf("\n1. Insert\n2. Lookup\n3. Display\n4. Exit\n");
        printf("Enter Choice: ");
        scanf("%d", &choice);

        switch(choice)
        {
            case 1: insert();
                    break;

            case 2: lookup();
                    break;

            case 3: display();
                    break;

            case 4: return 0;

            default: printf("Invalid Choice!\n");
        }
    }
}
